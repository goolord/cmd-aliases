alias=IF "$*" == "" (doskey /macros) ELSE (IF /I "$1" == "SAVE" (doskey /macros $g "%USERPROFILE%\doskey.txt" & ECHO Aliases SAVED) ELSE (IF /I "$1"=="LOAD" (doskey /macrofile="%USERPROFILE%\doskey.txt" & ECHO Aliases LOADED & doskey /macros) ELSE (IF /I "$1" == "help" (ECHO SAVE, LOAD, 'alias example= command', open %USERPROFILE%\txt to see list of aliases ) ELSE (doskey $*))))
